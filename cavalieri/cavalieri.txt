Ketinit();
Setparent(Cdyname()+"fig");

Setwindow([-3,5],[-3,5]);

Slider("S",[0,-4],[8,-4]);
s=S.x/2;
fun1="x+2";
fun2="x^2";
Plotdata("1",fun1,"x",["Num=1"]);
Plotdata("2",fun2,"x");
n=30;
forall(1..n,
 x=-1+#/n*3;
 y1=parse(Assign(fun1,["x",x]));
 y2=parse(Assign(fun2,["x",x]));
 Listplot(text(#)+"o",[[x,y1],[x,y2]],["dr,3","Color=gray"]);
 dy=y1-y2;
 y1=y1-s;
 if(y1<0,y1=0);
 y2=y1-dy;
 Listplot(text(#)+"d",[[x,y1],[x,y2]],["dr,3","Color=red"]);
);

Figpdf();
Windispg();
